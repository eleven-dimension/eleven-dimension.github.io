#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <ctime>
#include <queue>

using namespace std;

struct Oper
{
    int tim, op, ind;
    bool operator < (const Oper& ano) const
    {
        return tim > ano.tim;
    }
};

priority_queue<Oper> q;
queue<Oper> f;

void print(priority_queue<Oper> qq)
{
    cout << "---------------------------------" << endl;
    while (qq.size())
    {
        cout << qq.top().tim << " ";
        qq.pop();
    }
    cout << endl;
    cout << "-----------------------------" << endl;
}

void print(queue<Oper> qq)
{
    cout << "----------------------------" << endl;
    while (qq.size())
    {
        cout << qq.front().tim << " ";
        qq.pop();
    }
    cout << endl;
    cout << "--------------------" << endl;
}

int main()
{
    srand(unsigned(time(NULL)));

    for (int i = 1; i <= 5; i++)
    {
        int ti = rand() % 20 + 1;
        int op = rand() % 2 + 1;
        q.push({ ti,op,i });
    }

    //q.push({ 1,1,1 });

    queue<Oper> fl;
    while (q.size())

        if (fl.size() == 0 && q.size())

            auto tp = q.top();
            q.pop();
            if (tp.op == 1)

                cout << "Time: " << tp.tim << "  " << "Plane " << tp.ind << " begs to fly" << endl;

                fl.push({ tp.tim + 5,1,tp.ind });
            }
            else
            {
                cout << "Time: " << tp.tim << "  " << "Plane " << tp.ind << " begs to land" << endl;
                cout << "Time: " << tp.tim << "  " << "Plane " << tp.ind << " begins to land" << endl;
                fl.push({ tp.tim + 3,2,tp.ind });
            }
        }
        while (q.size() && fl.size() && q.top().tim <= fl.front().tim)
        {
            auto tp = q.top();
            q.pop();
            if (tp.op == 1)
            {
                cout << "Time: " << tp.tim << "  " << "Plane " << tp.ind << " begs to fly" << endl;
                fl.push({ fl.back().tim + 5,1,tp.ind });
            }
            else
            {
                cout << "Time: " << tp.tim << "  " << "Plane " << tp.ind << " begs to land" << endl;
                fl.push({ fl.back().tim + 3,2,tp.ind });
            }
        }

        //print(q);

        if (q.size())

                {
                    cout << "Time: " << fl.front().tim << "  " << "Plane " << fl.front().ind << " has flied away" << endl;

                else
                {
                    cout << "Time: " << fl.front().tim << "  " << "Plane " << fl.front().ind << " has landed" << endl;

                fl.pop();
                if (fl.size())

                        cout << "Time: " << cpy.tim << "  " << "Plane " << fl.front().ind << " begins to fly" << endl;
                    }
                    else
                    {
                        cout << "Time: " << cpy.tim << "  " << "Plane " << fl.front().ind << " begins to land" << endl;
